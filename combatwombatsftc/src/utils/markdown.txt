"use client";

import renderMathInElement from "https://cdn.jsdelivr.net/npm/katex@0.16.10/dist/contrib/auto-render.mjs";
import "katex/dist/katex.min.css";
import React, { useRef, useEffect } from "react";

export default function Markdown({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  const katexTextRef = useRef(null);
  useEffect(() => {
    if (katexTextRef.current) {
      renderMathInElement(katexTextRef.current, {
        delimiters: [
          { left: "$$", right: "$$", display: true },
          { left: "$", right: "$", display: false },
        ],
      });
    }
  }, []);

  return <div ref={katexTextRef}>{children}</div>;
}
